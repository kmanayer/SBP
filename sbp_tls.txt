(defprotocol sbp basic
  (defrole client
    (vars (cred answer data) (enc_cookie mesg) (c p name))
    (trace (send (enc "login:" cred (privk c)))
      (recv (cat "login-successful" enc_cookie))
      (send (cat "request" enc_cookie))
      (recv (enc "answer:" answer (privk p)))))
  (defrole proxy
    (vars (cc id s cred iv cookie answer sskey data) (c p name))
    (trace (recv cc) (send (cat id (pubk p))) (recv (enc s (pubk p)))
      (recv (enc id (hash s cc id))) (send (enc cc (hash s cc id)))
      (recv (enc (enc "login:" cred (privk c)) (hash s cc id)))
      (send
        (enc "login-successful" iv
          (enc cookie (hash sskey (hash s cc id))) (hash s cc id)))
      (recv
        (enc "request" iv (enc cookie (hash sskey (hash s cc id)))
          (hash s cc id)))
      (send (enc (enc "answer:" answer (privk p)) (hash s cc id))))
    (non-orig sskey)
    (uniq-gen id)))
